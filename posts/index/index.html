Spirng MVC 源码分析(一)

13/Apr 2016 By 王庆

从一个单线程程序切换为多线程程序在给我们带来好处的同时也会产生一些额外的成本，不要因为会使用多线程就将一个程序变为多线程实现。在准备使用多线程时，我们应该有一个清楚的认识：使用多线程带来的好处大于其成本，当有不确定时，我们应该尝试度量应用程序的性能和响应性来决定是否采用多线程，而不是靠猜来决定。

@RequestMapping(value = "/listSpecifyPushMsgs", method = RequestMethod.GET)
public JsonResult<SpecifyMsgs> listSpecifyPushMsgs(@RequestParam("lastTime") long lastTime, @RequestParam("size") int size, HttpServletRequest request) {
    long userId = (long) request.getAttribute(Constants.REQUEST_USER_ID_KEY);
    JsonResult<SpecifyMsgs> result = new JsonResult<>();
    if (lastTime < 0 || size < 1 || size > Constants.MAX_LIST_PAGE_ITEM_SIZE) {
        result.setResultCode(ResultCode.PARAMS_ERROR);
            return result;
    }
    
    SpecifyMsgs specifyMsgs = new SpecifyMsgs();
    specifyMsgs.setHasMore(false);
    List<SpecifyPush> specifyPushes = pushService.listSpecifyPushMsgs(userId, PushType.SPECIFY_PUSH_LH_AND_HCG_RECOMMEND, lastTime == 0 ? System.currentTimeMillis() : lastTime, size + 1);
    if (!CollectionUtils.isEmpty(specifyPushes)) {
        specifyMsgs.setHasMore(specifyPushes.size() > size);
        int count = 0;
        List<SpecifyMsgs.SpecifyMsg> specifyMsgList = new ArrayList<>();
        for (SpecifyPush specifyPush : specifyPushes) {
            if (count < size) {
                specifyMsgList.add(new SpecifyMsgs.SpecifyMsg(specifyPush.getTitle(),
                        specifyPush.getAlert(), specifyPush.getType(), specifyPush.getTime()));
            } else {
                break;
            }
                count++;
            }
            specifyMsgs.setSpecifyMsgs(specifyMsgList);
        }
        result.setData(specifyMsgs);
        return result;
    }
